361 static inline signed int scale(mad_fixed_t sample) // @suppress("Unused static function")
362 {
363  /* round */
364  sample += (1L << (MAD_F_FRACBITS - 16));
365 
366  /* clip */
367  if (sample >= MAD_F_ONE)
368  sample = MAD_F_ONE - 1;
369  else if (sample < -MAD_F_ONE)
370  sample = -MAD_F_ONE;
371 
372  /* quantize */
373  return sample >> (MAD_F_FRACBITS + 1 - 16);
374 }
